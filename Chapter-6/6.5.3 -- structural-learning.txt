from pomegranate import BayesianNetwork
import seaborn, time, numpy, matplotlib
seaborn.set_style('whitegrid')

from sklearn.datasets import load_digits
X, y = load_digits(10, True)

tic = time.time()
model = BayesianNetwork.from_samples(X) 
t = time.time() - tic
p = model.log_probability(X).sum()
print("Greedy")
print("Time (s): ", t)
print("P(D|M): ", p)
model.plot('model-greedy')

tic = time.time()
model = BayesianNetwork.from_samples(X, algorithm='exact-dp') 
t = time.time() - tic
p = model.log_probability(X).sum()
print("exact-dp")
print("Time (s): ", t)
print("P(D|M): ", p)
model.plot('model-exact-dp')

tic = time.time()
model = BayesianNetwork.from_samples(X, algorithm='exact') 
t = time.time() - tic
p = model.log_probability(X).sum()
print("exact")
print("Time (s): ", t)
print("P(D|M): ", p)
model.plot('model-exact')
